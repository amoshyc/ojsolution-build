#include <bits/stdc++.h>
using namespace std;

int K, A, B, V;

int main() {
    scanf("%d %d %d %d", &K, &A, &B, &V);

    int cnt = 0;
    int divi = B;

    int ans = -1;
    for (int box = 1; ; box++) {
        int sec = 1;
        if (divi >= K) {
            sec = K;
            divi -= K - 1;
        }
        else if (divi > 0) {
            sec = divi + 1;
            divi = 0;
        }

        cnt += sec * V;

        // printf("%d: %d, %d\n", box, sec, cnt);

        if (cnt >= A) {
            ans = box;
            break;
        }
    }

    printf("%d\n", ans);

    return 0;
}
