#include <iostream>
#include <algorithm>
#include <vector>
#include <cstdio>

using namespace std;

int main() {
    // ios::sync_with_stdio(false);

    int N;
    while (scanf("%d", &N)) {
        if (N == 0) break;

        int max_ = 0;
        int max_ending_here = 0;
        for (int i = 0; i < N; i++) {
            int inp;
            scanf("%d", &inp);

            max_ending_here = max(0, max_ending_here + inp);
            max_ = max(max_, max_ending_here);
        }

        printf("%d\n", max_);
    }

    return 0;
}
